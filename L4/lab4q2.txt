CREATE TABLE S as SELECT * FROM public.s;
CREATE TABLE P as SELECT * FROM public.p;

ALTER TABLE S ADD CONSTRAINT PK_SNO PRIMARY KEY (SNO);
ALTER TABLE P ADD CONSTRAINT PK_PNO PRIMARY KEY (PNO);

CREATE TABLE J(jno VARCHAR(5) PRIMARY KEY NOT NULL,jname VARCHAR(25) NOT NULL, city VARCHAR (25) NOT NULL);

CREATE TABLE SPJ(SNO VARCHAR, PNO VARCHAR, JNO VARCHAR, QTY INTEGER, PRIMARY KEY(SNO,PNO,JNO),FOREIGN KEY (SNO) REFERENCES S(SNO), FOREIGN KEY (PNO) REFERENCES P(PNO), FOREIGN KEY (JNO) REFERENCES J(JNO));
